```
import requests
from bs4 import BeautifulSoup

def check_phishing(url):
    try:
        # –ü–æ–ª—É—á–∞–µ–º —Å–æ–¥–µ—Ä–∂–∏–º–æ–µ –≤–µ–±-—Å—Ç—Ä–∞–Ω–∏—Ü—ã
        response = requests.get(url)
        response.raise_for_status()

        # –ê–Ω–∞–ª–∏–∑–∏—Ä—É–µ–º —Å–æ–¥–µ—Ä–∂–∏–º–æ–µ —Å –ø–æ–º–æ—â—å—é BeautifulSoup
        soup = BeautifulSoup(response.text, 'html.parser')

        # –ü—Ä–∏–º–µ—Ä: –ü—Ä–æ–≤–µ—Ä—è–µ–º –Ω–∞–ª–∏—á–∏–µ –∫–ª—é—á–µ–≤–æ–≥–æ —Å–ª–æ–≤–∞ "—Ñ–∏—à–∏–Ω–≥" –≤ –∑–∞–≥–æ–ª–æ–≤–∫–µ
        phishing_keyword = '—Ñ–∏—à–∏–Ω–≥'
        if phishing_keyword.lower() in soup.title.text.lower():
            print(f"–ü—Ä–µ–¥—É–ø—Ä–µ–∂–¥–µ–Ω–∏–µ: –ó–∞–≥–æ–ª–æ–≤–æ–∫ –≤–µ–±-—Å—Ç—Ä–∞–Ω–∏—Ü—ã —Å–æ–¥–µ—Ä–∂–∏—Ç '{phishing_keyword}'. –≠—Ç–æ –º–æ–∂–µ—Ç –±—ã—Ç—å —Ñ–∏—à–∏–Ω–≥–æ–≤—ã–π —Å–∞–π—Ç.")
        else:
            print("–í–µ–±-—Å—Ç—Ä–∞–Ω–∏—Ü–∞ –∫–∞–∂–µ—Ç—Å—è –±–µ–∑–æ–ø–∞—Å–Ω–æ–π.")

    except requests.exceptions.RequestException as e:
        print(f"–û—à–∏–±–∫–∞: {e}")

if __name__ == "__main__":
    # –ó–∞–ø—Ä–æ—Å–∏—Ç–µ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è –≤–≤–µ—Å—Ç–∏ URL –¥–ª—è –ø—Ä–æ–≤–µ—Ä–∫–∏
    user_url = input("–í–≤–µ–¥–∏—Ç–µ URL –¥–ª—è –ø—Ä–æ–≤–µ—Ä–∫–∏ –Ω–∞ —Ñ–∏—à–∏–Ω–≥: ")
    check_phishing(user_url)
```
